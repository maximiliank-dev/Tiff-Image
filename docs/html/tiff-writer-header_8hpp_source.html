119 void write_ifd_element(TiffTagType tag, TiffDataVariant variant, std::vector<TiffDataVariant>& rational_values) {
120 TiffDataType type = get_tag_data_type(tag);
121
122 // write the IFD
123 //tag
124 ushort_t tag_short = static_cast<ushort_t>(tag);
125 std::array<char, 2> tag_array = this->_endian_handler->convert_to_array(tag_short);
126 write_char<2>(tag_array, this->_stream);
127
128 //data type
129 std::array<char, 2> type_array = this->_endian_handler->convert_to_array(static_cast<ushort_t>(type));
130 write_char<2>(type_array, this->_stream);
131
132#if PRINT == 1
133 std::cout << "tag " << int(tag) << " " << to_string(tag) << std::endl;
134#endif
135 // value (in offset)
136 std::visit([this, &rational_values](auto& el) -> void {
137 using T = std::decay_t<decltype(el)>;
138
139 // std::string must be handles separate
140 if constexpr ( std::is_same<T, std::string>::value) {
141 throw std::runtime_error("string not implemented yet");
142 } else if constexpr(std::is_same<T, std::vector<uint8_t>>::value ||
143 std::is_same<T, std::vector<ushort_t>>::value ||
144 std::is_same<T, std::vector<uint_t>>::value ) { //use the implemented array conversion function
145
146
147 if(el.size() == 0) {
148 throw std::runtime_error("Error: TiffDataVariant's vector length is 0");
149 } else if(el.size() == 1) {
150 //data count
151 std::array<char, 4> tag_count = this->_endian_handler->convert_to_array(static_cast<uint_t>(1));
152 write_char<4>(tag_count, this->_stream);
153
154 auto data = el.at(0);
155 // convert the value to int
156 uint_t offset_value = static_cast<uint_t>(data);
157 std::array<char, 4> array = this->_endian_handler->convert_to_array(offset_value);
158 write_char<4>(array, this->_stream);
159 } else {
160 //data count
161 std::array<char, 4> tag_count = this->_endian_handler->convert_to_array(static_cast<uint_t>(el.size()));
162 write_char<4>(tag_count, this->_stream);
163
164 //write the offset to the TIFF IDF
165 std::array<char, 4> array = this->_endian_handler->convert_to_array(
166 static_cast<uint_t>(this->_offset) );
167 write_char<4>(array, this->_stream);
168
169 using element_type = T::value_type;
170 const size_t type_size = sizeof(T::value_type);
171 rational_values.push_back(el);
172
173 //move the next entry towards typesize * number of elements written
174 this->_offset += type_size*el.size();
175 }
176 } else if constexpr(std::is_same<T, std::vector<uint64_t>>::value) {
177 //data count
178 std::array<char, 4> tag_count = this->_endian_handler->convert_to_array(static_cast<uint_t>(el.size()));
179 write_char<4>(tag_count, this->_stream);
180
181 //write the offset to the TIFF IDF
182 std::array<char, 4> array = this->_endian_handler->convert_to_array(
183 static_cast<uint_t>(this->_offset) );
184 write_char<4>(array, this->_stream);
185
186 rational_values.push_back(el);
187
188 //move the next entry towards typesize * number of elements written
189 this->_offset += sizeof(uint64_t)*el.size();
190 } else {
191 static_assert("Error not implemented");
192 throw std::runtime_error("Type not implemented");
193 }
194 }, variant);
195 }

200 void write_IFDs() {
201 //setup the set of TIFF IFD entries
202 this->tags_to_set();
203
204 //backup the beginning of the IFD
205 std::streampos offset_tiff_data = this->_stream.tellp();
206
207 std::vector<TiffDataVariant> rational_values;
208 const size_t ifd_cnt = this->_values.size();
209
210 // add the number of bytes of the IFD size STRIP offset and StripByteCount
211 // also add the "Next IFD offset"(4 Bytes) byte count and the Number of "Directory Entries"(2 bytes)
212 // unit is in 'number of bytes'
213 // Formula: start position + (number of ifd)*Tag Size + (Directory Entries) + NumIFD(2) + NextIFD(4)
214 this->_offset = static_cast<uint64_t>(offset_tiff_data) + (ifd_cnt) * TiffTag_size +2+4; //182 = 32 + 12*12 + 2 + 4
215
216 // write the IDF count to the file
217 std::array<char, 2 > tmp = this->_endian_handler->convert_to_array(
218 static_cast<ushort_t>(ifd_cnt) );
219 write_char<2>( tmp, this->_stream );
220
221 // add the offset of the image data StripOffsets
222 this->_values[TiffTagType::StripOffsets] = make_variant(TiffTagType::StripOffsets, this->_offset_data);
223
224 // write the rest
225 for(auto it = this->_values.begin(); it != this->_values.end(); it++ ) {
226 this->write_ifd_element(it->first, it->second, rational_values);
227 }
228
229 // write the next IDF offset (is 0 in this case)
230 std::array<char, 4 > tmp4 = this->_endian_handler->convert_to_array(uint_t{0x0});
231 write_char<4>( tmp4, this->_stream );
232
233 for(auto el : rational_values) {
234 std::visit( [this](auto& vec) {
235 using T = std::decay_t<decltype(vec)>;
236 using element_type = T::value_type;
237 for(auto element : vec) {
238
239 if constexpr(std::is_same_v<element_type, char>) {
240 std::array<char, 1 > tmp = {element};
241 write_char<1>( tmp, this->_stream );
242 }
243 else if constexpr(std::is_same_v<element_type, uint8_t>) {
244 std::array<char, 1 > tmp = this->_endian_handler->convert_to_array((element));
245 write_char<1>( tmp, this->_stream );
246 }
247 else if constexpr(std::is_same_v<element_type, ushort_t>) {
248 std::array<char, 2 > tmp = this->_endian_handler->convert_to_array(element);
249 write_char<2>( tmp, this->_stream );
250 }
251 else if constexpr(std::is_same_v<element_type, uint_t>) {
252 std::array<char, 4 > tmp = this->_endian_handler->convert_to_array(element);
253 write_char<4>( tmp, this->_stream );
254 }
255 else if constexpr(std::is_same_v<element_type, uint64_t>) {
256 std::array<char, 8 > tmp = this->_endian_handler->convert_to_array(element);
257 write_char<8>( tmp, this->_stream );
258 }
259 }
260 }, el);
261 }
262
263 // set the pointer to the IFD Offset in the header
264 this->_stream.seekp(0x4);
265 std::array<char, 4 > offset_idf_header = this->_endian_handler->convert_to_array(static_cast<uint_t>(offset_tiff_data));
266 write_char<4>(offset_idf_header, this->_stream);
267 this->_stream.seekp(std::ios::app);
268 }

273 void write_image_data() {
274
275 const std::vector<TP>& data = this->_img->get_data();
276 // compute the size of a row
277 const size_t num_pixels = this->_img->get_pixel_number_of_colors();
278 const size_t size_row = this->_img->get_width()*num_pixels;
279
280 constexpr TP max = std::numeric_limits<TP>::max();
281
282 for(size_t i = 0; i < this->_img->get_height(); i++ ) {
283 for(size_t j = 0; j < this->_img->get_width(); j++) {
284 for(size_t k = 0; k < this->_img->get_pixel_number_of_colors(); k++) {
285 //check if the value range is inverted in a Grayscale or Bitlevel image
286 if( this->_values[TiffTagType::PhotometricInterpretation] ==
287 make_variant(TiffTagType::PhotometricInterpretation, {0x0}) ) {
288 TP diff = max - data[i*size_row + j*num_pixels*+k];
289 auto array = this->_endian_handler->convert_to_array(diff);
290 write_char(array, this->_stream);
291 } else {
292 TP dat = data[i*size_row + j*num_pixels+k];
293 auto array = this->_endian_handler->convert_to_array(dat);
294 write_char(array, this->_stream);
295 }
296 }
297 }
298 }
