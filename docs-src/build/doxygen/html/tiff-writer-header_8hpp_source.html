131 void write_ifd_element(TiffTagType tag, TiffDataVariant variant,
132 std::vector<TiffDataVariant>& rational_values) {
133 TiffDataType type = get_tag_data_type(tag);
134
135 // write the IFD
136 // tag
137 ushort_t tag_short = static_cast<ushort_t>(tag);
138 std::array<char, 2> tag_array =
139 this->_endian_handler->convert_to_array(tag_short);
140 write_char<2>(tag_array, this->_stream);
141
142 // data type
143 std::array<char, 2> type_array =
144 this->_endian_handler->convert_to_array(
145 static_cast<ushort_t>(type));
146 write_char<2>(type_array, this->_stream);
147
148#if PRINT == 1
149 std::cout << "tag " << int(tag) << " " << to_string(tag) << std::endl;
150#endif
151 // value (in offset)
152 std::visit(
153 [this, &rational_values](auto& el) -> void {
154 using T = std::decay_t<decltype(el)>;
155
156 // std::string must be handles separate
157 if constexpr (std::is_same<T, std::string>::value) {
158 throw std::runtime_error("string not implemented yet");
159 } else if constexpr (
160 std::is_same<T, std::vector<uint8_t>>::value ||
161 std::is_same<T, std::vector<ushort_t>>::value ||
162 std::is_same<T, std::vector<uint_t>>::
163 value) { // use the implemented array conversion
164 // function
165
166 if (el.size() == 0) {
167 throw std::runtime_error(
168 "Error: TiffDataVariant's vector length is 0");
169 } else if (sizeof(typename T::value_type) * el.size() <= 4) {
170 constexpr size_t elem_size = sizeof(typename T::value_type);
171
172 // data count
173 std::array<char, 4> tag_count =
174 this->_endian_handler->convert_to_array(
175 static_cast<uint_t>(el.size()));
176 write_char<4>(tag_count, this->_stream);
177
178 // write all values inline
179 for (auto& val : el) {
180 if constexpr (std::is_same_v<typename T::value_type, uint8_t>) {
181 write_char<1>(this->_endian_handler->convert_to_array(
182 static_cast<uint8_t>(val)), this->_stream);
183 } else if constexpr (std::is_same_v<typename T::value_type, ushort_t>) {
184 write_char<2>(this->_endian_handler->convert_to_array(
185 static_cast<ushort_t>(val)), this->_stream);
186 } else {
187 write_char<4>(this->_endian_handler->convert_to_array(
188 static_cast<uint_t>(val)), this->_stream);
189 }
190 }
191 // pad remaining bytes to fill the 4-byte Value/Offset field
192 for (size_t i = elem_size * el.size(); i < 4; ++i) {
193 write_char('\0', this->_stream);
194 }
195 } else {
196 // data count
197 std::array<char, 4> tag_count =
198 this->_endian_handler->convert_to_array(
199 static_cast<uint_t>(el.size()));
200 write_char<4>(tag_count, this->_stream);
201
202 // write the offset to the TIFF IDF
203 std::array<char, 4> array =
204 this->_endian_handler->convert_to_array(
205 static_cast<uint_t>(this->_offset));
206 write_char<4>(array, this->_stream);
207
208 using element_type = T::value_type;
209 const size_t type_size = sizeof(T::value_type);
210 rational_values.push_back(el);
211
212 // move the next entry towards typesize * number of
213 // elements written
214 this->_offset += type_size * el.size();
215 }
216 } else if constexpr (std::is_same<T, std::vector<uint64_t>>::
217 value) {
218 // data count
219 std::array<char, 4> tag_count =
220 this->_endian_handler->convert_to_array(
221 static_cast<uint_t>(el.size()));
222 write_char<4>(tag_count, this->_stream);
223
224 // write the offset to the TIFF IDF
225 std::array<char, 4> array =
226 this->_endian_handler->convert_to_array(
227 static_cast<uint_t>(this->_offset));
228 write_char<4>(array, this->_stream);
229
230 rational_values.push_back(el);
231
232 // move the next entry towards typesize * number of elements
233 // written
234 this->_offset += sizeof(uint64_t) * el.size();
235 } else {
236 static_assert("Error not implemented");
237 throw std::runtime_error("Type not implemented");
238 }
239 },
240 variant);
241 }

246 void write_IFDs() {
247 // setup the set of TIFF IFD entries
248 this->tags_to_set();
249
250 // Override StripByteCounts with the actual written byte count
251 // (tags_to_set uses the uncompressed size; after compression it may differ)
252 this->_values[TiffTagType::StripByteCounts] =
253 make_variant(TiffTagType::StripByteCounts, {this->_strip_byte_count});
254
255 // backup the beginning of the IFD
256 std::streampos offset_tiff_data = this->_stream.tellp();
257
258 std::vector<TiffDataVariant> rational_values;
259 const size_t ifd_cnt = this->_values.size();
260
261 // add the number of bytes of the IFD size STRIP offset and
262 // StripByteCount also add the "Next IFD offset"(4 Bytes) byte count and
263 // the Number of "Directory Entries"(2 bytes) unit is in 'number of
264 // bytes' Formula: start position + (number of ifd)*Tag Size +
265 // (Directory Entries) + NumIFD(2) + NextIFD(4)
266 this->_offset = static_cast<uint64_t>(offset_tiff_data) +
267 (ifd_cnt)*TiffTag_size + 2 +
268 4; // 182 = 32 + 12*12 + 2 + 4
269
270 // write the IDF count to the file
271 std::array<char, 2> tmp = this->_endian_handler->convert_to_array(
272 static_cast<ushort_t>(ifd_cnt));
273 write_char<2>(tmp, this->_stream);
274
275 // add the offset of the image data StripOffsets
276 this->_values[TiffTagType::StripOffsets] =
277 make_variant(TiffTagType::StripOffsets, this->_offset_data);
278
279 // write the rest
280 for (auto it = this->_values.begin(); it != this->_values.end(); it++) {
281 this->write_ifd_element(it->first, it->second, rational_values);
282 }
283
284 // write the next IDF offset (is 0 in this case)
285 std::array<char, 4> tmp4 =
286 this->_endian_handler->convert_to_array(uint_t{0x0});
287 write_char<4>(tmp4, this->_stream);
288
289 for (auto el : rational_values) {
290 std::visit(
291 [this](auto& vec) {
292 using T = std::decay_t<decltype(vec)>;
293 using element_type = T::value_type;
294 for (auto& element : vec) {
295 if constexpr (std::is_same_v<element_type, char>) {
296 std::array<char, 1> tmp = {element};
297 write_char<1>(tmp, this->_stream);
298 } else if constexpr (std::is_same_v<element_type,
299 uint8_t>) {
300 std::array<char, 1> tmp =
301 this->_endian_handler->convert_to_array(
302 (element));
303 write_char<1>(tmp, this->_stream);
304 } else if constexpr (std::is_same_v<element_type,
305 ushort_t>) {
306 std::array<char, 2> tmp =
307 this->_endian_handler->convert_to_array(
308 element);
309 write_char<2>(tmp, this->_stream);
310 } else if constexpr (std::is_same_v<element_type,
311 uint_t>) {
312 std::array<char, 4> tmp =
313 this->_endian_handler->convert_to_array(
314 element);
315 write_char<4>(tmp, this->_stream);
316 } else if constexpr (std::is_same_v<element_type,
317 uint64_t>) {
318 std::array<char, 8> tmp =
319 this->_endian_handler->convert_to_array(
320 element);
321 write_char<8>(tmp, this->_stream);
322 }
323 }
324 },
325 el);
326 }
327
328 // set the pointer to the IFD Offset in the header
329 this->_stream.seekp(0x4);
330 std::array<char, 4> offset_idf_header =
331 this->_endian_handler->convert_to_array(
332 static_cast<uint_t>(offset_tiff_data));
333 write_char<4>(offset_idf_header, this->_stream);
334 }

356 void compress(std::vector<TP>& data) {
357 if (this->_values[TiffTagType::Compression] ==
358 make_variant(TiffTagType::Compression,
359 static_cast<uint64_t>(TiffCompression::PackBits))) {
360 data = compression::PackBits::compress(data);
361 } else if (this->_values[TiffTagType::Compression] ==
362 make_variant(TiffTagType::Compression,
363 static_cast<uint64_t>(TiffCompression::LZW))) {
364 std::vector<uint16_t> codes = compression::LZW::compress(data);
365 data.clear();
366 for (uint16_t code : codes) {
367 std::array<char, 2> arr = this->_endian_handler->convert_to_array(code);
368 data.push_back(static_cast<uint8_t>(arr[0]));
369 data.push_back(static_cast<uint8_t>(arr[1]));
370 }
371 }
